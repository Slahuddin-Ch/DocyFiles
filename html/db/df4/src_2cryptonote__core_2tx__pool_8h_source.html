65  class txCompare
66  {
67  public:
68  bool operator()(const tx_by_fee_and_receive_time_entry& a, const tx_by_fee_and_receive_time_entry& b) const
69  {
70  // sort by greatest first, not least
71  if (a.first.first > b.first.first) return true;
72  else if (a.first.first < b.first.first) return false;
73  else if (a.first.second < b.first.second) return true;
74  else if (a.first.second > b.first.second) return false;
75  else if (a.second != b.second) return true;
76  else return false;
77  }
78  };
