60  static inline void check_rpc_cost(rpc_payment_state_t &rpc_payment_state, const char *call, uint64_t post_call_credits, uint64_t pre_call_credits, double expected_cost)
61  {
62  uint64_t expected_credits = (uint64_t)expected_cost;
63  if (expected_credits == 0)
64  expected_credits = 1;
65 
66  rpc_payment_state.credits = post_call_credits;
67  rpc_payment_state.expected_spent += expected_credits;
68 
69  if (pre_call_credits <= post_call_credits)
70  return;
71 
72  uint64_t cost = pre_call_credits - post_call_credits;
73 
74  if (cost == expected_credits)
75  {
76  MDEBUG("Call " << call << " cost " << cost << " credits");
77  return;
78  }
79  MWARNING("Call " << call << " cost " << cost << " credits, expected " << expected_credits);
80 
81  if (cost > expected_credits)
82  {
83  uint64_t d = cost - expected_credits;
84  if (rpc_payment_state.discrepancy > std::numeric_limits<uint64_t>::max() - d)
85  {
86  MERROR("Integer overflow in credit discrepancy calculation, setting to max");
87  rpc_payment_state.discrepancy = std::numeric_limits<uint64_t>::max();
88  }
89  else
90  {
91  rpc_payment_state.discrepancy += d;
92  }
93  }
94  }
