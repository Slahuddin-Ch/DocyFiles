50  {

53  // load
54  template <template <bool> class Archive>
55  bool do_serialize(Archive<false>& ar)
56  {
57  // size - 1 - because of variant tag
58  for (size = 1; size <= TX_EXTRA_PADDING_MAX_COUNT; ++size)
59  {
60  if (ar.eof())
61  break;
62 
63  uint8_t zero;
64  if (!::do_serialize(ar, zero))
65  return false;
66 
67  if (0 != zero)
68  return false;
69  }
70 
71  return size <= TX_EXTRA_PADDING_MAX_COUNT;
72  }
73 
74  // store
75  template <template <bool> class Archive>
76  bool do_serialize(Archive<true>& ar)
77  {
78  if(TX_EXTRA_PADDING_MAX_COUNT < size)
79  return false;
80 
81  // i = 1 - because of variant tag
82  for (size_t i = 1; i < size; ++i)
83  {
84  uint8_t zero = 0;
85  if (!::do_serialize(ar, zero))
86  return false;
87  }
88  return true;
89  }
90  };
