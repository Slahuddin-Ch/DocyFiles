372  void update(uint64_t height, bool force = false)
373  {
374  auto current_time = std::chrono::system_clock::now();
375  const auto node_update_threshold = std::chrono::seconds(DIFFICULTY_TARGET_V1 / 2); // use min of V1/V2
376  if (node_update_threshold < current_time - m_blockchain_height_update_time || m_blockchain_height <= height)
377  {
378  update_blockchain_height();
379  m_blockchain_height = (std::max)(m_blockchain_height, height);
380  }
381 
382  if (std::chrono::milliseconds(20) < current_time - m_print_time || force)
383  {
384  std::cout << QT_TRANSLATE_NOOP("cryptonote::simple_wallet", "Height ") << height << " / " << m_blockchain_height << '\r' << std::flush;
385  m_print_time = current_time;
386  }
387  }
388 
389  private:
390  void update_blockchain_height()
391  {
392  std::string err;
393  uint64_t blockchain_height = m_simple_wallet.get_daemon_blockchain_height(err);
394  if (err.empty())
395  {
396  m_blockchain_height = blockchain_height;
397  m_blockchain_height_update_time = std::chrono::system_clock::now();
398  }
399  else
400  {
401  LOG_ERROR("Failed to get current blockchain height: " << err);
402  }
403  }
