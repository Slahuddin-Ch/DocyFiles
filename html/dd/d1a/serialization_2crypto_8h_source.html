42 template <template <bool> class Archive>
43 bool do_serialize(Archive<false> &ar, std::vector<crypto::signature> &v)
44 {
45  size_t cnt = v.size();
46  v.clear();
47 
48  // very basic sanity check
49  if (ar.remaining_bytes() < cnt*sizeof(crypto::signature)) {
50  ar.set_fail();
51  return false;
52  }
53 
54  v.reserve(cnt);
55  for (size_t i = 0; i < cnt; i++) {
56  v.resize(i+1);
57  ar.serialize_blob(&(v[i]), sizeof(crypto::signature), "");
58  if (!ar.good())
59  return false;
60  }
61  return true;
62 }
