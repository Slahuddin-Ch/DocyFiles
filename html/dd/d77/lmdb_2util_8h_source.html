112  inline int less(MDB_val const* left, MDB_val const* right) noexcept
113  {
114  if (!left || !right || left->mv_size < sizeof(T) + offset || right->mv_size < sizeof(T) + offset)
115  {
116  assert("invalid use of custom comparison" == 0);
117  return -1;

119 
120  T left_val;
121  T right_val;
122  std::memcpy(std::addressof(left_val), static_cast<char*>(left->mv_data) + offset, sizeof(T));
123  std::memcpy(std::addressof(right_val), static_cast<char*>(right->mv_data) + offset, sizeof(T));
124  return left_val < right_val ? -1 : bool(right_val < left_val);
125  }
