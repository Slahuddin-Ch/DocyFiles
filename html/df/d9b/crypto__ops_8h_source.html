113  case op_addKeys: rct::addKeys(key, point0, point1); break;
114  case op_scalarmultBase: rct::scalarmultBase(scalar0); break;
115  case op_scalarmultKey: rct::scalarmultKey(point0, scalar0); break;
116  case op_scalarmultH: rct::scalarmultH(scalar0); break;

121  case op_ge_triple_scalarmult_base_vartime: ge_triple_scalarmult_base_vartime(&tmp_p2, scalar0.bytes, scalar1.bytes, precomp1, scalar2.bytes, precomp2); break;
