76 template<typename T>
77 void Combinator<T>::doCombine(size_t from, size_t k)
78 {
79  current.push_back(0);
80 
81  for (size_t i = from; i <= origin.size() - k; ++i)
82  {

85  if (k > 1) {
86  doCombine(i + 1, k - 1);

92  combinations.push_back(comb);
