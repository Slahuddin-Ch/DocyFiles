328  template<bool W, template <bool> class Archive>
329  bool serialize_rctsig_base(Archive<W> &ar, size_t inputs, size_t outputs)
330  {
331  FIELD(type)
332  if (type == RCTTypeNull)
333  return ar.good();
334  if (type != RCTTypeFull && type != RCTTypeSimple && type != RCTTypeBulletproof && type != RCTTypeBulletproof2 && type != RCTTypeCLSAG && type != RCTTypeBulletproofPlus)
335  return false;
336  VARINT_FIELD(txnFee)
337  // inputs/outputs not saved, only here for serialization help
338  // FIELD(message) - not serialized, it can be reconstructed
339  // FIELD(mixRing) - not serialized, it can be reconstructed
340  if (type == RCTTypeSimple) // moved to prunable with bulletproofs
341  {
342  ar.tag("pseudoOuts");
343  ar.begin_array();
344  PREPARE_CUSTOM_VECTOR_SERIALIZATION(inputs, pseudoOuts);
345  if (pseudoOuts.size() != inputs)
346  return false;
347  for (size_t i = 0; i < inputs; ++i)
348  {
349  FIELDS(pseudoOuts[i])
350  if (inputs - i > 1)
351  ar.delimit_array();
352  }
353  ar.end_array();
354  }
355 
356  ar.tag("ecdhInfo");
357  ar.begin_array();
358  PREPARE_CUSTOM_VECTOR_SERIALIZATION(outputs, ecdhInfo);
359  if (ecdhInfo.size() != outputs)
360  return false;
361  for (size_t i = 0; i < outputs; ++i)
362  {
363  if (type == RCTTypeBulletproof2 || type == RCTTypeCLSAG || type == RCTTypeBulletproofPlus)
364  {
365  ar.begin_object();
366  if (!typename Archive<W>::is_saving())
367  memset(ecdhInfo[i].amount.bytes, 0, sizeof(ecdhInfo[i].amount.bytes));
368  crypto::hash8 &amount = (crypto::hash8&)ecdhInfo[i].amount;
369  FIELD(amount);
370  ar.end_object();
371  }
372  else
373  {
374  FIELDS(ecdhInfo[i])
375  }
376  if (outputs - i > 1)
377  ar.delimit_array();
378  }
379  ar.end_array();
380 
381  ar.tag("outPk");
382  ar.begin_array();
383  PREPARE_CUSTOM_VECTOR_SERIALIZATION(outputs, outPk);
384  if (outPk.size() != outputs)
385  return false;
386  for (size_t i = 0; i < outputs; ++i)
387  {
388  FIELDS(outPk[i].mask)
389  if (outputs - i > 1)
390  ar.delimit_array();
391  }
392  ar.end_array();
393  return ar.good();
394  }

396  BEGIN_SERIALIZE_OBJECT()
397  FIELD(type)
398  FIELD(message)
399  FIELD(mixRing)
400  FIELD(pseudoOuts)
401  FIELD(ecdhInfo)
402  FIELD(outPk)
403  VARINT_FIELD(txnFee)
404  END_SERIALIZE()
405  };

414  // when changing this function, update cryptonote::get_pruned_transaction_weight
415  template<bool W, template <bool> class Archive>
416  bool serialize_rctsig_prunable(Archive<W> &ar, uint8_t type, size_t inputs, size_t outputs, size_t mixin)
417  {
418  if (inputs >= 0xffffffff)
419  return false;
420  if (outputs >= 0xffffffff)
421  return false;
422  if (mixin >= 0xffffffff)
423  return false;
424  if (type == RCTTypeNull)
425  return ar.good();
426  if (type != RCTTypeFull && type != RCTTypeSimple && type != RCTTypeBulletproof && type != RCTTypeBulletproof2 && type != RCTTypeCLSAG && type != RCTTypeBulletproofPlus)
427  return false;
428  if (type == RCTTypeBulletproofPlus)
429  {
430  uint32_t nbp = bulletproofs_plus.size();
431  VARINT_FIELD(nbp)
432  ar.tag("bpp");
433  ar.begin_array();
434  if (nbp > outputs)
435  return false;
436  PREPARE_CUSTOM_VECTOR_SERIALIZATION(nbp, bulletproofs_plus);
437  for (size_t i = 0; i < nbp; ++i)
438  {
439  FIELDS(bulletproofs_plus[i])
440  if (nbp - i > 1)
441  ar.delimit_array();
442  }
443  if (n_bulletproof_plus_max_amounts(bulletproofs_plus) < outputs)
444  return false;
445  ar.end_array();
446  }
447  else if (type == RCTTypeBulletproof || type == RCTTypeBulletproof2 || type == RCTTypeCLSAG)
448  {
449  uint32_t nbp = bulletproofs.size();
450  if (type == RCTTypeBulletproof2 || type == RCTTypeCLSAG)
451  VARINT_FIELD(nbp)
452  else
453  FIELD(nbp)
454  ar.tag("bp");
455  ar.begin_array();
456  if (nbp > outputs)
457  return false;
458  PREPARE_CUSTOM_VECTOR_SERIALIZATION(nbp, bulletproofs);
459  for (size_t i = 0; i < nbp; ++i)
460  {
461  FIELDS(bulletproofs[i])
462  if (nbp - i > 1)
463  ar.delimit_array();
464  }
465  if (n_bulletproof_max_amounts(bulletproofs) < outputs)
466  return false;
467  ar.end_array();
468  }
469  else
470  {
471  ar.tag("rangeSigs");
472  ar.begin_array();
473  PREPARE_CUSTOM_VECTOR_SERIALIZATION(outputs, rangeSigs);
474  if (rangeSigs.size() != outputs)
475  return false;
476  for (size_t i = 0; i < outputs; ++i)
477  {
478  FIELDS(rangeSigs[i])
479  if (outputs - i > 1)
480  ar.delimit_array();
481  }
482  ar.end_array();
483  }
484 
485  if (type == RCTTypeCLSAG || type == RCTTypeBulletproofPlus)
486  {
487  ar.tag("CLSAGs");
488  ar.begin_array();
489  PREPARE_CUSTOM_VECTOR_SERIALIZATION(inputs, CLSAGs);
490  if (CLSAGs.size() != inputs)
491  return false;
492  for (size_t i = 0; i < inputs; ++i)
493  {
494  // we save the CLSAGs contents directly, because we want it to save its
495  // arrays without the size prefixes, and the load can't know what size
496  // to expect if it's not in the data
497  ar.begin_object();
498  ar.tag("s");
499  ar.begin_array();
500  PREPARE_CUSTOM_VECTOR_SERIALIZATION(mixin + 1, CLSAGs[i].s);
501  if (CLSAGs[i].s.size() != mixin + 1)
502  return false;
503  for (size_t j = 0; j <= mixin; ++j)
504  {
505  FIELDS(CLSAGs[i].s[j])
506  if (mixin + 1 - j > 1)
507  ar.delimit_array();
508  }
509  ar.end_array();
510 
511  ar.tag("c1");
512  FIELDS(CLSAGs[i].c1)
513 
514  // CLSAGs[i].I not saved, it can be reconstructed
515  ar.tag("D");
516  FIELDS(CLSAGs[i].D)
517  ar.end_object();
518 
519  if (inputs - i > 1)
520  ar.delimit_array();
521  }
522 
523  ar.end_array();
524  }
525  else
526  {
527  ar.tag("MGs");
528  ar.begin_array();
529  // we keep a byte for size of MGs, because we don't know whether this is
530  // a simple or full rct signature, and it's starting to annoy the hell out of me
531  size_t mg_elements = (type == RCTTypeSimple || type == RCTTypeBulletproof || type == RCTTypeBulletproof2) ? inputs : 1;
532  PREPARE_CUSTOM_VECTOR_SERIALIZATION(mg_elements, MGs);
533  if (MGs.size() != mg_elements)
534  return false;
535  for (size_t i = 0; i < mg_elements; ++i)
536  {
537  // we save the MGs contents directly, because we want it to save its
538  // arrays and matrices without the size prefixes, and the load can't
539  // know what size to expect if it's not in the data
540  ar.begin_object();
541  ar.tag("ss");
542  ar.begin_array();
543  PREPARE_CUSTOM_VECTOR_SERIALIZATION(mixin + 1, MGs[i].ss);
544  if (MGs[i].ss.size() != mixin + 1)
545  return false;
546  for (size_t j = 0; j < mixin + 1; ++j)
547  {
548  ar.begin_array();
549  size_t mg_ss2_elements = ((type == RCTTypeSimple || type == RCTTypeBulletproof || type == RCTTypeBulletproof2) ? 1 : inputs) + 1;
550  PREPARE_CUSTOM_VECTOR_SERIALIZATION(mg_ss2_elements, MGs[i].ss[j]);
551  if (MGs[i].ss[j].size() != mg_ss2_elements)
552  return false;
553  for (size_t k = 0; k < mg_ss2_elements; ++k)
554  {
555  FIELDS(MGs[i].ss[j][k])
556  if (mg_ss2_elements - k > 1)
557  ar.delimit_array();
558  }
559  ar.end_array();
560 
561  if (mixin + 1 - j > 1)
562  ar.delimit_array();
563  }
564  ar.end_array();
565 
566  ar.tag("cc");
567  FIELDS(MGs[i].cc)
568  // MGs[i].II not saved, it can be reconstructed
569  ar.end_object();
570 
571  if (mg_elements - i > 1)
572  ar.delimit_array();
573  }
574  ar.end_array();
575  }
576  if (type == RCTTypeBulletproof || type == RCTTypeBulletproof2 || type == RCTTypeCLSAG || type == RCTTypeBulletproofPlus)
577  {
578  ar.tag("pseudoOuts");
579  ar.begin_array();
580  PREPARE_CUSTOM_VECTOR_SERIALIZATION(inputs, pseudoOuts);
581  if (pseudoOuts.size() != inputs)
582  return false;
583  for (size_t i = 0; i < inputs; ++i)
584  {
585  FIELDS(pseudoOuts[i])
586  if (inputs - i > 1)
587  ar.delimit_array();
588  }
589  ar.end_array();
590  }
591  return ar.good();
592  }
